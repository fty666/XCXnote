			inputeValue:''，
			list:['学习英文','学习react']
		}
	}
	
	render(){
		return{
			<input value="{this.state.inputeValue}"
			//会转化输入内容中的html标签，
			dangerouslySetInnerHTML={{__html:item}}
			onChange="{this.change.bind(this)}"></input>
		}
		change(e){
			const value=e.target.value;
			this.setState(()=>({
				inputeValue:value
			})
			)}
	}
